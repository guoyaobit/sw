136     class GraphTraversalState
137     {
138     public:
139         //  note: these state values are strictly ordered...
140         enum State { e_undiscovered = 0U, e_discovered = 1U, e_finishable = 2U, e_finished = 3U };
141     
142         GraphTraversalState() :
143             m_e(e_undiscovered), m_discovered(-1), m_finishable(-1), m_finished(-1) { }
144     
145         GraphTraversalState(const GraphTraversalState &o) :
146             m_e(o.m_e), m_discovered(o.m_discovered),
147             m_finishable(o.m_finishable), m_finished(o.m_finished) { }
148     
149         virtual ~GraphTraversalState() { }
150     
151         void setDiscovered(int t) { m_e = e_discovered; m_discovered = t; }
152         void setFinished(int t)   { m_e = e_finished;   m_finished   = t; }
153         void setFinishable(int t) { m_e = e_finishable; m_finishable = t; }
154     
155         bool isUndiscovered() const { return m_e == e_undiscovered; }
156         bool isDiscovered()   const { return m_discovered != -1; }
157         bool isFinished()     const { return m_finished   != -1; }
158         bool isFinishable()   const { return m_e == e_finishable; }
159     
160         State state()     const { return m_e; }
161         int discoveryTime()  const { return m_discovered; }
162         int finishTime()     const { return m_finished;   }
163         int finishableTime() const { return m_finishable; }
164     
165         const std::string toString()
166         {
167             std::stringstream ss;
168             const char * state_names[] = { "und", "dis", "fab", "fin" };
169             ss << "s=" << state_names[int(m_e)] << " dis=" << m_discovered << " fab=" << m_finishable << " fin=" << m_finished;
170             return ss.str();
171         }
172     
173         bool operator==(const GraphTraversalState &rhs) const
174         {
175             return (m_e == rhs.m_e) && (m_discovered == rhs.m_discovered) &&
176                 (m_finishable == rhs.m_finishable) && (m_finished == rhs.m_finished);
177         }
178     
179         bool operator<(const GraphTraversalState &rhs) const
180         {
181             if ( *this == rhs )
182             {
183                 return false;
184             }
185             // we know they won't be == now.
186     
187             // states: undiscovered < discovered < finishable < finished
188             // within each state (i.e. comparing two with same state) use associated time.
189             // but, unless absolutely == (all states' times the same) we should use prior
190             // states' times as well to determine order of two items in the same state.
191     
192             if ( int(m_e) != int(rhs.m_e) )
193             {
194                 // not in the same state just compare which state they're in
195                 // this isn't really interesting... typically we're comparing
196                 // states which are all of one type.  but do something reasonable.
197                 return int(m_e) < int(rhs.m_e);
198             }
199             // now we know they're both in the same state ( && != each other )
200             switch ( m_e )
201             {
202                 case e_discovered:
203                     return m_discovered < rhs.m_discovered;
204                     break;
205     
206                 case e_finishable:
207                     if ( m_finishable == rhs.m_finishable )
208                     {
209                         return (m_discovered < rhs.m_discovered);
210                     }
211                     else
212                     {
213                         return m_finishable < rhs.m_finishable;
214                     }
215                     break;
216     
217                 case e_finished:
218                     if ( m_finished == rhs.m_finished )
219                     {
220                         if ( m_finishable == rhs.m_finishable )
221                         {
222                             return m_discovered < rhs.m_discovered;
223                         }
224                         else
225                         {
226                             return m_finishable < rhs.m_finishable;
227                         }
228                     }
229                     else
230                     {
231                         return m_finished < rhs.m_finished;
232                     }
233                     break;
234     
235                 case e_undiscovered:
236                 default:
237                     return false; // should be == as there is no time.
238             }
239     
240             /*not reached*/
241             return false;
242         }
243     
244     protected:
245         State m_e;
246         int m_discovered;  // -1 := not yet discovered
247         int m_finishable;  // -1 := not yet finishable
248         int m_finished;    // -1 := not yet finished
249     };

305     template <class G>
306     class GraphScoreboard
307     {
308     public:
309         GraphScoreboard(G *g = 0) : m_graph(g) { }
310         virtual ~GraphScoreboard() { }
311     
312         typedef typename G::Node Node;
313         typedef typename G::Edge Edge;
314         typedef typename G::Elem Elem;
315         typedef typename G::ElemSequence ElemSequence;
316         typedef typename G::NodeSequence NodeSequence;
317         typedef typename G::EdgeSequence EdgeSequence;
318         typedef typename G::ElemSequenceIterator ElemSequenceIterator;
319         typedef typename G::NodeSequenceIterator NodeSequenceIterator;
320         typedef typename G::EdgeSequenceIterator EdgeSequenceIterator;
321     
322         typedef ast::GraphTraversalState Score;
323         typedef std::map<Node *, Score, typename G::nodeCompareFn> NodeScores;
324         typedef std::map<Edge *, Score, typename G::edgeCompareFn> EdgeScores;
325         typedef std::map<Elem, Score, typename G::elemCompareFn>   ElemScores;
326         typedef typename NodeScores::iterator NodeScoresIterator;
327         typedef typename EdgeScores::iterator EdgeScoresIterator;
328         typedef typename ElemScores::iterator ElemScoresIterator;
329     
330         NodeScoresIterator nodeBegin() { return m_node_score.begin(); }
331         NodeScoresIterator nodeEnd() { return m_node_score.end(); }
332         NodeScoresIterator findNode(Node *n) { return m_node_score.find(n); }
333     
334         EdgeScoresIterator edgeBegin() { return m_edge_score.begin(); }
335         EdgeScoresIterator edgeEnd() { return m_edge_score.end(); }
336         EdgeScoresIterator findEdge(Edge *e) { return m_edge_score.find(e); }
337     
338         ElemScoresIterator elemBegin() { return m_elem_score.begin(); }
339         ElemScoresIterator elemEnd() { return m_elem_score.end(); }
340         ElemScoresIterator findElem(Elem e) { return m_elem_score.find(e); }
341     
342         std::pair<NodeScoresIterator, bool> insertNode(std::pair<Node *, GraphTraversalState> p) { return m_node_score.insert(p); }
343         std::pair<EdgeScoresIterator, bool> insertEdge(std::pair<Edge *, GraphTraversalState> p) { return m_edge_score.insert(p); }
344         std::pair<ElemScoresIterator, bool> insertElem(std::pair<Elem,   GraphTraversalState> p) { return m_elem_score.insert(p); }
345     
346         void clear() { m_node_score.clear(); m_edge_score.clear(); m_elem_score.clear(); }
347         void setGraph(G *g) { m_graph = g; clear(); }
348     
349         //
350         // fetch means find it and if it doesn't exist create it and mark as discovered
351         //
352         EdgeScoresIterator fetchEdgeScore(int time, Edge *edge)
353         {
354             // gLogInfo << __func__ << " edge=" << edge->id() << std::endl;
355             EdgeScoresIterator f_i = findEdge(edge);
356             if ( f_i == edgeEnd() )
357             {
358                 std::pair<EdgeScoresIterator, bool> inserted =
359                     insertEdge(std::pair<Edge *, GraphTraversalState>(edge, GraphTraversalState()));
360                 f_i = inserted.first;
361                 f_i->second.setDiscovered(time);
362             }
363             return f_i;
364         }
365     
366     
367         NodeScoresIterator fetchNodeScore(int time, Node *node)
368         {
369             // gLogInfo << __func__ << " node=" << node->id() << std::endl;
370             NodeScoresIterator f_i = findNode(node);
371             if ( f_i == nodeEnd() )
372             {
373                 std::pair<NodeScoresIterator, bool> inserted =
374                     insertNode(std::pair<Node *, GraphTraversalState>(node, GraphTraversalState()));
375                 f_i = inserted.first;
376                 f_i->second.setDiscovered(time);
377             }
378             return f_i;
379         }
380     
381         // the fetch elem score entry points are slightly different
382         // because they are derived from already-present (finished)
383         // nodes and edges.
384         ElemScoresIterator fetchElemScore(Node *node, Score score)
385         {
386             Elem elem(node, 0);
387             ElemScoresIterator f_i = findElem( elem );
388             // gLogInfo << __func__ << " node=" << node->id() << std::endl;
389             if ( f_i == elemEnd() )
390             {
391                 insertElem(std::pair<Elem, Score>(elem, score));
392             }
393             return f_i;
394         }
395     
396         ElemScoresIterator fetchElemScore(Edge *edge, Score score)
397         {
398             Elem elem(0, edge);
399             ElemScoresIterator f_i = findElem( elem );
400             // gLogInfo << __func__ << " node=" << node->id() << std::endl;
401             if ( f_i == elemEnd() )
402             {
403                 insertElem(std::pair<Elem, Score>(elem, score));
404             }
405             return f_i;
406         }
407     
408         //
409         // evaluate edge state based upon upstream nodes.
410         // this can result in an edge being *finished*.
411         // node evaluation doesn't do that.
412         //
413         EdgeScoresIterator evaluateEdgeScore(int time, Edge *edge)
414         {
415             // gLogInfo << __func__ << " edge=" << edge->id() << std::endl;
416             EdgeScoresIterator edge_score_i;
417             NodeSequence up_nodes;
418     
419             edge_score_i = fetchEdgeScore(time, edge);
420     
421             if ( edge_score_i->second.isFinished() )
422             {
423                 return edge_score_i;
424             }
425     
426             up_nodes = m_graph->upstreamNodes(edge);
427     
428             bool all_nodes_finished = true; // note: edges w/o upstream nodes are finishable at 0...
429             int nodes_finishable_time = 0;
430             int nodes_finished_time   = 0;
431     
432             for ( size_t un_i = 0, UN_I = up_nodes.size(); un_i != UN_I; ++un_i )
433             {
434                 Node *up_node = up_nodes[un_i];
435                 NodeScoresIterator node_score_i;
436     
437                 node_score_i = fetchNodeScore(time, up_node);
438     
439                 all_nodes_finished = all_nodes_finished && node_score_i->second.isFinished();
440                 nodes_finished_time   = std::max<int>(nodes_finished_time, node_score_i->second.finishTime());
441                 nodes_finishable_time = std::max<int>(nodes_finishable_time, node_score_i->second.finishableTime());
442             }
443     
444             if ( all_nodes_finished )
445             {
446                 edge_score_i->second.setFinishable(nodes_finishable_time); // := same as node finishable
447                 edge_score_i->second.setFinished(nodes_finished_time); // := same as node finish
448     
449                 // gLogInfo << "finish edge=" << edge_score_i->first->id() << " score=" << edge_score_i->second.toString() << std::endl;
450                 fetchElemScore(edge_score_i->first, edge_score_i->second);
451             }
452     
453             return edge_score_i;
454         }
455     
456     
457         //
458         // evaluate node state based upon upstream edges.
459         // unlike edge evaluation, node evaluation does *not* finish nodes.
460         //
461         NodeScoresIterator evaluateNodeScore(int time, Node *node)
462         {
463             // gLogInfo << __func__ << " node=" << node->id() << std::endl;
464             NodeScoresIterator node_score_i;
465     
466             node_score_i = fetchNodeScore(time, node);
467     
468             if ( node_score_i->second.isFinished() )
469             {
470                 return node_score_i;
471             }
472     
473             // input edges
474             EdgeSequence up_edges = m_graph->upstreamEdges(node);
475     
476             bool all_up_edges_finishable = true; // nodes w/o upstream edges are finishable.
477             int all_up_edges_finishable_time = 0;
478     
479             for (EdgeSequenceIterator ue_i = up_edges.begin(); all_up_edges_finishable && (ue_i != up_edges.end()); ++ue_i )
480             {
481                 EdgeScoresIterator up_edge_score_i;
482     
483                 up_edge_score_i = evaluateEdgeScore(time, *ue_i);
484     
485                 all_up_edges_finishable = all_up_edges_finishable && (up_edge_score_i->second.isFinished() || up_edge_score_i->second.isFinishable());
486                 all_up_edges_finishable_time = std::max<int>(all_up_edges_finishable_time, up_edge_score_i->second.finishableTime());
487             }
488     
489             if ( all_up_edges_finishable )
490             {
491                 node_score_i->second.setFinishable(all_up_edges_finishable_time + 1); // := last edge completion + 1
492             }
493     
494             return node_score_i;
495         }
496     
497     
498         NodeScoresIterator evaluateScore(GraphTraversalPointer<G> &ptr, int t)
499         {
500             NodeScoresIterator f_i = evaluateNodeScore(t, ptr.node());
501             ptr.setState(f_i->second);
502             return f_i;
503         }
504     
505         //
506         // finishes a node and increments time. note: any upstream edges which were finishable
507         // but which hadn't been yet take time == t and the node itself takes time = t + 1;
508         //
509         NvDlaError finishNode(GraphTraversalPointer<G> &ptr, int t)
510         {
511             NvDlaError e = NvDlaError_Success;
512     
513             EdgeSequence up_edges = m_graph->upstreamEdges(ptr.node());
514             for ( EdgeSequenceIterator up_i = up_edges.begin(); up_i != up_edges.end(); ++up_i)
515             {
516                 EdgeScoresIterator edge_score_i = fetchEdgeScore(t, *up_i);
517                 if ( edge_score_i->second.isFinishable() ) // implies !edge_score_i->second.isFinished() )
518                 {
519                     edge_score_i->second.setFinished(t);
520                     // gLogInfo << "at node=" << ptr.node()->id() << " finish edge=" << (*up_i)->id() << " score=" << edge_score_i->second.toString() << std::endl;
521                     fetchElemScore(edge_score_i->first, edge_score_i->second);
522                 }
523             }
524     
525             NodeScoresIterator score_i = fetchNodeScore(t, ptr.node());
526             score_i->second.setFinished(t+1);
527             ptr.setState(score_i->second);
528     
529             fetchElemScore(score_i->first, score_i->second);
530     
531             return e;
532         }
533     
534     protected:
535         G *m_graph;
536         NodeScores m_node_score;
537         EdgeScores m_edge_score;
538         ElemScores m_elem_score;
539     };

545     template <class G>
546     class ScoredGraphOrdering
547     {
548     public:
549     
550         typedef typename G::Node Node;
551         typedef typename G::Edge Edge;
552         typedef typename G::Elem Elem;
553     
554         typedef typename G::NodeSequence NodeSequence;
555         typedef typename G::EdgeSequence EdgeSequence;
556         typedef typename G::ElemSequence ElemSequence;
557     
558         typedef typename G::NodeSequenceIterator NodeSequenceIterator;
559         typedef typename G::EdgeSequenceIterator EdgeSequenceIterator;
560         typedef typename G::ElemSequenceIterator ElemSequenceIterator;
561     
562         typedef typename GraphScoreboard<G>::Score Score;
563         typedef typename GraphScoreboard<G>::NodeScores NodeScores;
564         typedef typename GraphScoreboard<G>::EdgeScores EdgeScores;
565         typedef typename GraphScoreboard<G>::ElemScores ElemScores;
566     
567         typedef typename GraphScoreboard<G>::NodeScoresIterator NodeScoresIterator;
568         typedef typename GraphScoreboard<G>::EdgeScoresIterator EdgeScoresIterator;
569         typedef typename GraphScoreboard<G>::ElemScoresIterator ElemScoresIterator;
570     
571         ScoredGraphOrdering(G *graph = 0) : m_graph(graph), m_scores(graph) { }
572         virtual ~ScoredGraphOrdering() { }
573     
574         G *graph() const { return m_graph; }
575         void setGraph(G *graph)
576         {
577             clear();
578             m_graph = graph;
579         }
580     
581         GraphScoreboard<G> &scores() { return m_scores; }
582     
583         NodeScoresIterator findNodeScore(Node *n) { return m_scores.findNode(n); }
584         EdgeScoresIterator findEdgeScore(Edge *e) { return m_scores.findEdge(e); }
585         ElemScoresIterator findElemScore(Elem e)  { return m_scores.findElem(e); }
586     
587         const std::vector<ElemScoresIterator> &elemScoreOrder() const { return m_elem_score_order; }
588         const std::vector<NodeScoresIterator> &nodeScoreOrder() const { return m_node_score_order; }
589         const std::vector<EdgeScoresIterator> &edgeScoreOrder() const { return m_edge_score_order; }
590     
591         //
592         // the real business... generally speaking, does the following:
593         //     . seed the scheme with the nodes reachable from the graph's input edges
594         //     . while traversal pointers exist
595         //         . evaluate nodes at and edges immediately upstream of the pointers
596         //         . scrub finished pointers
597         //         . find first pointer which is finishable
598         //             . finish the node there
599         //             . advance pointer downstream
600         //                 . add more pointers as needed
601         //     . finish off dangling edges
602         //     . sort the scoreboard results to provide a linear ordering
603         //
604         // note that this scheme allows for portions of the graph to go undiscovered
605         // (or unfinished) if the input edges to the graph don't cover enough.
606         //
607         virtual NvDlaError generate() // default is DepthFirstDependencyOrder()
608         {
609             NvDlaError e = NvDlaError_Success;
610             int time = 0;
611             typename EdgeSequence::const_iterator ie_i;
612             std::list<GraphTraversalPointer<G>> pointer_list;
613     
614             // gLogInfo << "generating scoreboard." << std::endl;
615     
616             if ( !m_graph ) {
617                 ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "missing graph");
618             }
619     
620             clear();
621     
622             // for all graph input edges...
623             for ( ie_i = m_graph->inputEdges().begin(); ie_i != m_graph->inputEdges().end(); ++ie_i)
624             {
625                 // this *should* be zero sized, check it.
626                 NodeSequence upstream_nodes = m_graph->upstreamNodes(*ie_i);
627                 if ( upstream_nodes.size() )
628                 {
629                     ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unexpected source node for network input edge '%s'",
630                                          (*ie_i)->id().c_str());
631                 }
632                 NodeSequence downstream_nodes = m_graph->downstreamNodes(*ie_i);
633     
634                 // for all downstream nodes...
635                 for ( size_t ni = 0, NI = downstream_nodes.size(); ni != NI; ++ni )
636                 {
637                     Node *downstream_node = downstream_nodes[ni];
638                     GraphTraversalPointer<G> downstream_pointer(m_graph, downstream_node);
639                     // don't push more than once as there can be multiple paths to a node from the inputs.
640                     // also note that is the very first such time this happens and so it's impossible for
641                     // the GTP<G> pointers to have anything other than "discovered" as their current state.
642                     // that means the find GTP<G> (downstream_pointer) comparison is really only testing for
643                     // the presence of the same node.
644                     if ( pointer_list.end() == std::find(pointer_list.begin(), pointer_list.end(), downstream_pointer) )
645                     {
646                         pointer_list.push_back( downstream_pointer );
647                     }
648                 }
649             }
650     
651             while ( pointer_list.size() )
652             {
653                 // helper predicate
654                 struct GraphTraversalPointer_is_Finished
655                 {
656                     bool operator() (const GraphTraversalPointer<G> & tptr) { return tptr.state().isFinished(); }
657                 };
658     
659                 //
660                 // evaluate to find the first which is finishable
661                 //
662                 typename std::list<GraphTraversalPointer<G>>::iterator finishable_i;
663     
664                 finishable_i = pointer_list.end();
665     
666                 for ( typename std::list<GraphTraversalPointer<G>>::iterator p_i = pointer_list.begin();
667                       p_i != pointer_list.end();
668                       ++p_i)
669                 {
670                     NodeScoresIterator check_score_i;
671     
672                     check_score_i = m_scores.evaluateScore(*p_i, time);
673     
674                     if ( check_score_i->second.isFinishable() && (finishable_i == pointer_list.end() ) )
675                     {
676                         finishable_i = p_i;
677                     }
678                 }
679     
680                 //
681                 // scrub finished pointers from the list.
682                 // note: list iterators which aren't removed stay valid.
683                 //
684                 pointer_list.remove_if( GraphTraversalPointer_is_Finished() );
685     
686                 if ( pointer_list.size() == 0 )
687                 {
688                     continue; // done.
689                 }
690     
691                 //
692                 // something should be finishable every time through this list
693                 //
694                 if ( finishable_i == pointer_list.end() )
695                 {
696                     sort(); // just so the end result is quasi-legit.  don't ignore failure though.
697                     ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "%s is not converging for :%s %s ", __func__, pointer_list.front().node()->id().c_str(), pointer_list.front().node()->name().c_str() );
698                 }
699     
700                 GraphTraversalPointer<G> &ptr = *finishable_i;
701     
702                 //
703                 // finish and advance downstream
704                 //
705                 m_scores.finishNode(ptr, time);
706                 time++;
707     
708                 //
709                 // if there's more than one downstream node then add to the pointer list.
710                 // but add next to this one so stable-ish ordering is maintained.
711                 //
712                 NodeSequence advance_nodes;
713                 EdgeSequence down_edges = m_graph->downstreamEdges(ptr.node());
714     
715                 // for each downstream edge...
716                 for ( EdgeSequenceIterator de_i = down_edges.begin(); de_i != down_edges.end(); ++de_i )
717                 {
718                     Edge *down_edge = *de_i;
719                     NodeSequence down_nodes  = m_graph->downstreamNodes(down_edge);
720     
721                     // make sure any downstream dangling (no output nodes) edges get finished
722                     m_scores.evaluateEdgeScore(time, down_edge);
723     
724                     // for each downstream node on the (multi-)edge...
725                     for ( size_t ni = 0, NI = down_nodes.size(); ni != NI; ++ni ) {
726                         Node *down_node = down_nodes[ni];
727     
728                         // add the nodes to the list only once...
729                         if ( advance_nodes.end() == std::find(advance_nodes.begin(), advance_nodes.end(), down_node) )
730                         {
731                             advance_nodes.push_back(down_node);
732                         }
733                     }
734                 }
735     
736                 if ( advance_nodes.size() )
737                 {
738     #if 0
739                     gLogInfo << "\tadvance nodes:";
740                     for ( size_t ai = 0, AI = advance_nodes.size(); ai != AI; ++ai ) {
741                         gLogInfo << "[" << advance_nodes[ai]->id() << "] ";
742                     }
743                     gLogInfo << std::endl;
744     #endif
745     
746                     //
747                     // the first of these can be followed by our current pointer.  the rest should be inserted.
748                     // do the insertion in reverse so we can just use finishable_i as the insertion point.
749                     //
750                     int new_pointers = advance_nodes.size();
751                     typename std::list<GraphTraversalPointer<G>>::iterator insertion_point = ++finishable_i;
752     
753                     for ( int new_pointer_i = new_pointers - 1; new_pointer_i > 0; --new_pointer_i )
754                     {
755                         pointer_list.insert( insertion_point, GraphTraversalPointer<G>(m_graph, advance_nodes[new_pointer_i] ) );
756                     }
757                     ptr.setNode(advance_nodes[0]); // we'll reevaluate next time through...
758                 }
759     
760     #if 0
761                 gLogInfo << "\t ptr list: ";
762                 for ( typename std::list<GraphTraversalPointer<G>>::iterator ai = pointer_list.begin(); ai != pointer_list.end(); ++ai )
763                 {
764                     gLogInfo << "[" << ai->node()->id() << "] ";
765                 }
766                 gLogInfo << std::endl;
767     #endif
768             } // while path pointers exist
769     
770     
771             sort();
772     
773         fail:
774             return e;
775     
776         }

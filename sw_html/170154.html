47     void SDPActivationOpNode::captureCanonicalParams()
48     {
49         NvDlaError e = NvDlaSuccess;
50     
51         switch(canonicalNode()->params().activationType())
52         {
53         case ActivationType::kRELU:
54             params().x1Params().setEnabled(true);
55             params().x1Params().setActType(SDPActTypeEnum::SDP_ACT_TYPE_RELU);
56             break;
57         case ActivationType::kSIGMOID:
58             if ( !graph()->target_config()->isSDPLutCapable() )
59                 ORIGINATE_ERROR_FAIL(NvDlaError_NotSupported, "Activation type not supported: %d", canonicalNode()->params().activationType());
60     
61             params().yParams().setEnabled(true);
62             params().yParams().setActType(SDPActTypeEnum::SDP_ACT_TYPE_SIGMOID);
63     
64             // Register sigmoid lut with the LutManager
65             PROPAGATE_ERROR_FAIL(graph()->lutManager()->registerSigmoid(graph()->profile()->computePrecision().e(), &m_hLut));
66             break;
67         case ActivationType::kTANH:
68             if ( !graph()->target_config()->isSDPLutCapable() )
69                 ORIGINATE_ERROR_FAIL(NvDlaError_NotSupported, "Activation type not supported: %d", canonicalNode()->params().activationType());
70     
71             params().yParams().setEnabled(true);
72             params().yParams().setActType(SDPActTypeEnum::SDP_ACT_TYPE_TANH);
73     
74             // Register tanh lut with the LutManager
75             PROPAGATE_ERROR_FAIL(graph()->lutManager()->registerTanh(graph()->profile()->computePrecision().e(), &m_hLut));
76             break;
77         default:
78             params().x1Params().setActType(SDPActTypeEnum::SDP_ACT_TYPE_UNKNOWN);
79             ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unknown activation type: %d", canonicalNode()->params().activationType());
80         }
81     
82     fail:
83         return;
84     }

86     void engine_ast::SDPActivationOpNode::inheritParams(Node* otherNode)
87     {
88         // inherit the parameters that make sense (keep adding later)
89         SDPActivationOpNode* otherAct = NodeFactory::nodeCast<SDPActivationOpNode*>(otherNode);
90         m_hLut = otherAct->m_hLut;
91     
92         SDPNode* asSdpNode = NodeFactory::nodeCast<SDPNode*>(this);
93         SDPNode* otherActAsSdpNode = NodeFactory::nodeCast<SDPNode*>(otherNode);
94         asSdpNode->params().setConvMode(otherActAsSdpNode->params().convMode());
95         asSdpNode->params().setWinogradParams(otherActAsSdpNode->params().winogradParams());
96         asSdpNode->params().setNumGroups(otherActAsSdpNode->params().numGroups());
97     
98         params().setOutCVT(otherAct->params().outCVT());
99         params().setX1Params(otherAct->params().x1Params());
100     }

102     NvDlaError SDPActivationOpNode::emitReLU(DLASDPOpDescAccessor sdp_op)
103     {
104         NvDlaError e = NvDlaSuccess;
105     
106         DLASDPOpAccessor x1_op_acc = sdp_op.x1OpAccessor();
107     
108         *x1_op_acc.enable()      = 1;
109         *x1_op_acc.ALUType()     = x1_op_acc.ALUType_Sum();
110         *x1_op_acc.type()        = x1_op_acc.type_None();
111         *x1_op_acc.mode()        = x1_op_acc.mode_PerLayer();
112         *x1_op_acc.act()         = x1_op_acc.act_RelU();
113         *x1_op_acc.shiftValue()  = 0;
114         *x1_op_acc.ALUOperand()  = 0;
115         *x1_op_acc.MulOperand()  = 1;
116         *x1_op_acc.truncate()    = 0;
117         *x1_op_acc.precision()   = *sdp_op.srcPrecision();  // precision of engine = precision of its input tensor
118     
119         return e;
120     }

122     NvDlaError SDPActivationOpNode::emitLut(DLASDPOpDescAccessor sdp_op)
123     {
124         NvDlaError e = NvDlaSuccess;
125     
126         DLASDPOpAccessor y_op_acc = sdp_op.yOpAccessor();
127     
128         *sdp_op.LUTIndex()      = graph()->lutManager()->getIndex(m_hLut);
129     
130         *y_op_acc.enable()      = 1;
131         *y_op_acc.ALUType()     = y_op_acc.ALUType_Sum();
132         *y_op_acc.type()        = y_op_acc.type_None();
133         *y_op_acc.mode()        = y_op_acc.mode_PerLayer();
134         *y_op_acc.act()         = y_op_acc.act_LUT();
135         *y_op_acc.shiftValue()  = 0;
136         *y_op_acc.ALUOperand()  = 0;
137         *y_op_acc.MulOperand()  = 1;
138         *y_op_acc.truncate()    = 0;
139         *y_op_acc.precision()   = *sdp_op.srcPrecision();  // precision of engine = precision of its input tensor
140     
141         return e;
142     }

144     NvDlaError SDPActivationOpNode::emitOp(Graph *g,
145                                         DLAInterface *target_dla,
146                                         NvU32 op_slot, NvU32 batch_id,
147                                         DLACommonOpDescAccessor       dep,
148                                         DLAOperationContainerAccessor op,
149                                         DLASurfaceContainerAccessor   surf)
150     {
151         NvDlaError e = NvDlaSuccess;
152     
153         DLASDPOpDescAccessor  sdp_op = op.sdpOpDescAccessor(0);
154         DLACVTParamAccessor out_cvt_acc = sdp_op.outCVTAccessor();
155         DLASDPOpAccessor x1_op_acc = sdp_op.x1OpAccessor();
156         DLASDPOpAccessor x2_op_acc = sdp_op.x2OpAccessor();
157         DLASDPOpAccessor y_op_acc  = sdp_op.yOpAccessor();
158         DLASDPSurfaceDescAccessor surf_acc = surf.sdpSurfaceDescAccessor(0);
159         DLADataCubeAccessor src_data_acc    = surf_acc.srcDataAccessor();
160         DLADataCubeAccessor dst_data_acc    = surf_acc.dstDataAccessor();
161         DLAConsumerAccessor fused_acc = dep.fusedParentAccessor();
162         NVDLA_UNUSED(fused_acc);
163     
164         surface::TensorSurfaceDesc *src_tsd     = g->nodeInputTensorSurface(this, 0, supportedInSurfCategories());
165         surface::TensorSurfaceDesc *dst_tsd     = g->nodeOutputTensorSurface(this, 0, supportedOutSurfCategories());
166     
167         *sdp_op.srcPrecision()   = ASTToDLAInterface::getSDPPrecision(target_dla, src_tsd->surfaceFormat().precision());
168         *sdp_op.dstPrecision()   = ASTToDLAInterface::getSDPPrecision(target_dla, dst_tsd->surfaceFormat().precision());
169         *sdp_op.LUTIndex()       = -1;
170         *sdp_op.batchNum()       = 1;
171         *sdp_op.batchStride()    = 0;
172     
173         *out_cvt_acc.scale()     = params().outCVT().scale();
174         *out_cvt_acc.truncate()  = params().outCVT().truncate();
175         *out_cvt_acc.offset()    = params().outCVT().offset();
176         *out_cvt_acc.enable()    = static_cast<NvU8>(params().outCVT().isEnable());
177     
178         *x1_op_acc.enable() = 0;
179         *x2_op_acc.enable() = 0;
180         *y_op_acc.enable()  = 0;
181     
182         if (params(batch_id).x1Params().enabled())
183         {
184             switch(params(batch_id).x1Params().actType().v())
185             {
186             case SDPActTypeEnum::SDP_ACT_TYPE_RELU:
187                 PROPAGATE_ERROR_FAIL(emitReLU(sdp_op));
188                 break;
189             default:
190                 ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unknown activation type: %d", params(batch_id).x1Params().actType().v());
191             }
192         }
193     
194         if (params(batch_id).yParams().enabled())
195         {
196             switch(params(batch_id).yParams().actType().v())
197             {
198             case SDPActTypeEnum::SDP_ACT_TYPE_SIGMOID:
199             case SDPActTypeEnum::SDP_ACT_TYPE_TANH:
200                 PROPAGATE_ERROR_FAIL(emitLut(sdp_op));
201                 break;
202             default:
203                 ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unknown activation type: %d", params(batch_id).yParams().actType().v());
204             }
205         }
206     
207         emitDependencyParams(target_dla, dep, batch_id);
208         setDataCubeAccessor(src_data_acc, src_tsd, IODirectionEnum::INPUT, batch_id);
209         setDataCubeAccessor(dst_data_acc, dst_tsd, IODirectionEnum::OUTPUT, batch_id);
210     
211         if ( params(batch_id).convMode().v() == ConvolutionModeEnum::CONV_WINOGRAD )
212         {
213             *sdp_op.convMode() = sdp_op.convMode_Winograd();
214     
215             *src_data_acc.width() = params().winogradParams().ioDims.w;
216             *src_data_acc.height() = params().winogradParams().ioDims.h;
217     
218             *dst_data_acc.width() = params().winogradParams().ioDims.w;
219             *dst_data_acc.height() = params().winogradParams().ioDims.h;
220         }
221         else if ( params(batch_id).convMode().v() == ConvolutionModeEnum::CONV_DIRECT )
222         {
223             *sdp_op.convMode() = sdp_op.convMode_Direct();
224         }
225         else
226         {
227             ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unsupported Conv mode for %s", name().c_str());
228         }
229     
230         if ( g->debugOps() )
231         {
232             gLogInfo << "SDP activation node @ op_slot = " << op_slot << " batch_id = " << batch_id << endl;
233             gLogInfo << "\tsrc precision " << (int)*sdp_op.srcPrecision() << endl;
234             gLogInfo << "\tdst precision " << (int)*sdp_op.dstPrecision() << endl;
235             gLogInfo << "\tx1 enable " << (int)*x1_op_acc.enable() << endl;
236             if (*x1_op_acc.enable())
237             {
238                 gLogInfo << "\tx1 precision " << (int)*x1_op_acc.precision() << endl;
239                 gLogInfo << "\tx1 aluType " << (int)*x1_op_acc.ALUType() << endl;
240                 gLogInfo << "\tx1 type " << (int)*x1_op_acc.type() << endl;
241                 gLogInfo << "\tx1 mode " << (int)*x1_op_acc.mode() << endl;
242                 gLogInfo << "\tx1 act " << (int)*x1_op_acc.act() << endl;
243                 gLogInfo << "\tx1 shiftValue " << (int)*x1_op_acc.shiftValue() << endl;
244                 gLogInfo << "\tx1 aluOperand " << (int)*x1_op_acc.ALUOperand() << endl;
245                 gLogInfo << "\tx1 mulOperand " << (int)*x1_op_acc.MulOperand() << endl;
246                 gLogInfo << "\tx1 truncate " << (int)*x1_op_acc.truncate() << endl;
247             }
248             gLogInfo << "\tx2 enable " << (int)*x2_op_acc.enable() << endl;
249             if (*x2_op_acc.enable())
250             {
251                 gLogInfo << "\tx2 precision " << (int)*x2_op_acc.precision() << endl;
252                 gLogInfo << "\tx2 aluType " << (int)*x2_op_acc.ALUType() << endl;
253                 gLogInfo << "\tx2 type " << (int)*x2_op_acc.type() << endl;
254                 gLogInfo << "\tx2 mode " << (int)*x2_op_acc.mode() << endl;
255                 gLogInfo << "\tx2 act " << (int)*x2_op_acc.act() << endl;
256                 gLogInfo << "\tx2 shiftValue " << (int)*x2_op_acc.shiftValue() << endl;
257                 gLogInfo << "\tx2 aluOperand " << (int)*x2_op_acc.ALUOperand() << endl;
258                 gLogInfo << "\tx2 mulOperand " << (int)*x2_op_acc.MulOperand() << endl;
259                 gLogInfo << "\tx2 truncate " << (int)*x2_op_acc.truncate() << endl;
260             }
261             gLogInfo << "\ty enable " << (int)*y_op_acc.enable() << endl;
262             if (*y_op_acc.enable())
263             {
264                 gLogInfo << "\ty precision " << (int)*y_op_acc.precision() << endl;
265                 gLogInfo << "\ty aluType " << (int)*y_op_acc.ALUType() << endl;
266                 gLogInfo << "\ty type " << (int)*y_op_acc.type() << endl;
267                 gLogInfo << "\ty mode " << (int)*y_op_acc.mode() << endl;
268                 gLogInfo << "\ty act " << (int)*y_op_acc.act() << endl;
269                 gLogInfo << "\ty shiftValue " << (int)*y_op_acc.shiftValue() << endl;
270                 gLogInfo << "\ty aluOperand " << (int)*y_op_acc.ALUOperand() << endl;
271                 gLogInfo << "\ty mulOperand " << (int)*y_op_acc.MulOperand() << endl;
272                 gLogInfo << "\ty truncate " << (int)*y_op_acc.truncate() << endl;
273             }
274             gLogInfo << "\tsrc tsd:" << src_tsd->id() << endl;
275             gLogInfo << "\tdst tsd:" << dst_tsd->id() << endl;
276             gLogInfo << "\tdependencyCount" << (int)*dep.dependencyCount() << endl;
277             gLogInfo << "\tsrc addr=" << *src_data_acc.address() << endl;
278             gLogInfo << "\tsrc type=" << (int)*src_data_acc.type() << endl;
279             gLogInfo << "\tconv_mode " << (int)*sdp_op.convMode() << endl;
280             gLogInfo << "\tsrc size " << *src_data_acc.size()    << endl;
281             gLogInfo << "\tsrc width " << *src_data_acc.width()   << endl;
282             gLogInfo << "\tsrc height " << *src_data_acc.height()   << endl;
283             gLogInfo << "\tsrc channel " << *src_data_acc.channel()  << endl;
284             gLogInfo << "\tsrc linestride " << *src_data_acc.lineStride() << endl;
285             gLogInfo << "\tsrc surfstride " << *src_data_acc.surfStride()  << endl;
286             gLogInfo << "\tdst addr=" << *dst_data_acc.address() << endl;
287             gLogInfo << "\tdst type=" << (int)*dst_data_acc.type() << endl;
288             gLogInfo << "\tdst size " << *dst_data_acc.size()    << endl;

305     NvDlaError engine_ast::SDPActivationOpNode::emitOp(NvU32 op_slot, NvU32 batch_id,
306                                                     DLAInterface* target_dla,
307                                                     DLACommonOpDescAccessor&        dep,
308                                                     DLAOperationContainerAccessor&  op,
309                                                     DLASurfaceContainerAccessor&    surf,
310                                                     nvdla_prototest_interface::Layer* protoLayer)
311     {
312         NvDlaError e = NvDlaSuccess;
313         NvU8 numConsumers = 0;
314     
315         DLASDPOpDescAccessor  sdp_op    = op.sdpOpDescAccessor(0);
316         DLACVTParamAccessor out_cvt_acc = sdp_op.outCVTAccessor();
317         DLASDPOpAccessor x1_op_acc      = sdp_op.x1OpAccessor();
318         DLASDPOpAccessor x2_op_acc      = sdp_op.x2OpAccessor();
319         DLASDPOpAccessor y_op_acc       = sdp_op.yOpAccessor();
320         DLASDPSurfaceDescAccessor surf_acc  = surf.sdpSurfaceDescAccessor(0);
321         DLADataCubeAccessor src_data_acc    = surf_acc.srcDataAccessor();
322         DLADataCubeAccessor dst_data_acc    = surf_acc.dstDataAccessor();
323         DLADataCubeAccessor x1_data_acc     = surf_acc.x1DataAccessor();
324         DLADataCubeAccessor x2_data_acc     = surf_acc.x2DataAccessor();
325         DLADataCubeAccessor y_data_acc      = surf_acc.yDataAccessor();
326         DLAConsumerAccessor fused_acc = dep.fusedParentAccessor();
327         NVDLA_UNUSED(out_cvt_acc);
328         NVDLA_UNUSED(x1_data_acc);
329         NVDLA_UNUSED(x2_data_acc);
330         NVDLA_UNUSED(y_data_acc);
331         NVDLA_UNUSED(batch_id);
332     
333         surface::TensorSurfaceDesc *src_tsd     = graph()->nodeInputTensorSurface(this, 0, supportedInSurfCategories());
334         surface::TensorSurfaceDesc *dst_tsd     = graph()->nodeOutputTensorSurface(this, 0, supportedOutSurfCategories());
335     
336         nvdla_prototest_interface::SDPOpDesc* protoSDPOpDesc        = protoLayer->mutable_op_config()->mutable_sdp_op();
337         nvdla_prototest_interface::SDPSurfaceDesc* protoSDPSurfDesc = protoLayer->mutable_surface()->mutable_sdp_surface();
338         nvdla_prototest_interface::SDPOp*          protoSDPX1OpDesc = protoSDPOpDesc->mutable_x1_op();
339         nvdla_prototest_interface::SDPOp*          protoSDPX2OpDesc = protoSDPOpDesc->mutable_x2_op();
340         nvdla_prototest_interface::SDPOp*          protoSDPYOpDesc  = protoSDPOpDesc->mutable_y_op();
341         nvdla_prototest_interface::DataCube* protoSrcDataCube       = protoSDPSurfDesc->mutable_src_data();
342         nvdla_prototest_interface::DataCube* protoDstDataCube       = protoSDPSurfDesc->mutable_dst_data();
343         nvdla_prototest_interface::DataPrecision protoSrcPrec, protoDstPrec;
344     
345         protoLayer->set_index(op_slot);
346         protoLayer->set_roi_index(0);
347         protoLayer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_SDP);
348         protoLayer->set_dependency_count(*dep.dependencyCount());
349     
350         /* consumers */
351         for (size_t c = 0; c < EngineType::num_elements(); c++)
352         {
353             NvS8 fw_op_index = ASTToDLAInterface::getEngineType(target_dla, c);
354             if ( fw_op_index < 0 )
355             {
356                 continue;
357             }
358     
359             DLAConsumerAccessor cons_acc = dep.consumerAccessor(fw_op_index);
360             if (*cons_acc.index() != -1) {
361                 numConsumers++;
362                 nvdla_prototest_interface::Consumer* protoConsumer = protoLayer->add_bottom();
363                 protoConsumer->set_index(*cons_acc.index());
364                 switch(c) {
365                     case EngineTypeEnum::BDMA : protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_BDMA); break;
366                     case EngineTypeEnum::CONVOLUTION : protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_CONV); break;
367                     case EngineTypeEnum::SDP  : protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_SDP); break;
368                     case EngineTypeEnum::PDP  : protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_PDP); break;
369                     case EngineTypeEnum::CDP  : protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_CDP); break;
370                     case EngineTypeEnum::RUBIK: protoConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_RUBIK); break;
371                     default: ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unrecognized consumer");
372                 }
373                 switch(dependencyParams().consumer(c).opEvent().v()) {
374                     case OperationEventTypeEnum::OP_CDMA_WEIGHT_DONE : protoConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_CDMA_WT_DONE); break;
375                     case OperationEventTypeEnum::OP_CDMA_DATA_DONE   : protoConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_CDMA_DT_DONE); break;
376                     case OperationEventTypeEnum::OP_COMPLETED        : protoConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_OP_COMPLETED); break;
377                     case OperationEventTypeEnum::OP_ENABLED          : protoConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_OP_ENABLED); break;
378                     case OperationEventTypeEnum::OP_PROGRAMMED       : protoConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_OP_PROGRAMMED); break;
379                     default: ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unrecognized consumer event");
380                 }
381             }
382         }
383     
384         /* fused node */
385         if (dependencyParams().fusedNode(engine_ast::IODirectionEnum::INPUT))
386         {
387             nvdla_prototest_interface::Consumer* protoFusedConsumer = protoLayer->mutable_fused();
388             protoFusedConsumer->set_index(*fused_acc.index());
389             protoFusedConsumer->set_event(nvdla_prototest_interface::Consumer_EventType::Consumer_EventType_OP_ENABLED);
390             switch(dependencyParams().fusedNode(engine_ast::IODirectionEnum::INPUT)->engineType().v()) {
391                 case EngineTypeEnum::CONVOLUTION: protoFusedConsumer->set_type(nvdla_prototest_interface::LayerType::DLA_OP_CONV); break;
392                 default: ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "SDP can have only Conv op as its fused partner on input side");
393             }
394         }
395     
396         switch(src_tsd->surfaceFormat().precision().v()) {
397             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_FP16  : protoSrcPrec = nvdla_prototest_interface::DataPrecision::PRECISION_FP16; break;
398             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_INT16 : protoSrcPrec = nvdla_prototest_interface::DataPrecision::PRECISION_INT16; break;
399             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_INT8  : protoSrcPrec = nvdla_prototest_interface::DataPrecision::PRECISION_INT8; break;
400             default: ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unrecognized src precision");
401         }
402     
403         switch(dst_tsd->surfaceFormat().precision().v()) {
404             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_FP16  : protoDstPrec = nvdla_prototest_interface::DataPrecision::PRECISION_FP16; break;
405             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_INT16 : protoDstPrec = nvdla_prototest_interface::DataPrecision::PRECISION_INT16; break;
406             case  surface::SurfacePrecisionEnum::NVDLA_PRECISION_INT8  : protoDstPrec = nvdla_prototest_interface::DataPrecision::PRECISION_INT8; break;
407             default: ORIGINATE_ERROR_FAIL(NvDlaError_BadParameter, "Unrecognized dst precision");
408         }
409     
410         protoSDPOpDesc->set_src_precision(protoSrcPrec);
411         protoSDPOpDesc->set_dst_precision(protoDstPrec);
